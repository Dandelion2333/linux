
#加颜色，行号，自动缩进
set hlsearch            "高亮度反白 
set backspace=2         "可随时用退格键删除 
set autoindent          "自劢缩排 
set ruler               "可显示最后一行癿状态 
set showmode            "左下角那一行癿状态 
set nu                  "可以在每一行癿最前面显示行号啦！ 
set bg=dark             "显示丌同癿底色色调 
syntax on               "迚行语法检验，颜色显示。
set tabstop=4       	"设置tab为四个空格"
set softtabstop=4
set shiftwidth=4
set expandtab
